import java.util.Arrays;


public class Ordenacion {
    
    public static int[] ordenarDescendente (int num1, int num2, int num3){
        int[] contenedor = new int[3];
        contenedor[0] = num1;
        contenedor[1] = num2;
        contenedor[2] = num3;
        Arrays.sort(contenedor);
        int aux = contenedor[0];
        contenedor[0] = contenedor[2];
        contenedor[2] = aux;
        return contenedor;
    }
    
}
